5.
SELECT e.employee_id, e.name, e.job_title, e.department,
SUM(sd.sales) as sum_sales FROM employees e JOIN sales_data sd ON sd.employee_id = e.employee_id
GROUP BY sd.employee_id ORDER BY sum_sales DESC LIMIT 5;

6.
SELECT e.name, e.department, e.job_title, AVG(e.salary) as avg_salary FROM employees e GROUP BY e.department
HAVING avg_salary > (SELECT AVG(e2.salary) as avg_salary2 FROM employees e2 GROUP BY e2.department ORDER BY avg_salary2 LIMIT 1);

7.
SELECT   employee_id,name, department, job_title, total_sales, RANK() OVER (ORDER BY total_sales DESC) as sales_rank
FROM (
    SELECT e.employee_id, e.name, e.department, e.job_title, SUM(sd.sales) as total_sales
    FROM employees e
    JOIN sales_data sd ON e.employee_id= sd.employee_id
    GROUP BY e.employee_id, e.name, e.department, e.job_title
    ORDER BY total_sales DESC
    LIMIT 5
) AS ranked_sales;

CREATE PROCEDURE GetNameByDepartment(IN department_name VARCHAR(255))
BEGIN
    SELECT e.employee_id, e.name, e.salary
    FROM employees e
    WHERE e.department = department_name;

    FROM employees e
    WHERE e.department = department_name;
